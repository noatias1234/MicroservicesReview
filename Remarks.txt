git init
	cli command to init new git

git status
	cli command to see all uncommited and unstaged changes

git branch --show
	show local current branch

git branch --all
	show all branches, local and all remotes

git branch -D <branch-name>
	delete local branch, named 'branch-name'

git clone <remote_url>
	clone remote git repository to local machine

git pull
	pull the changes in a breanch on a git to local machine
